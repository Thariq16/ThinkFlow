1. Architecture Overview
ThinkFlow is a Flutter PWA (Phase 1) and mobile app (Phase 2) backed by Firebase and Google Cloud. All AI inference runs server-side via Vertex AI. The client is a pure consumer of Firestore streams and callable Cloud Functions — it holds no secrets and enforces no business rules.
Core Principle
The Flutter client is dumb by design. It renders state and calls functions. Every AI call, plan check, and business rule lives in a Cloud Function. This makes the client easy to test, the security surface small, and the AI layer swappable.

3. Firestore Data Model
All collections are flat at the root level. Tasks are a subcollection under projects — not a top-level collection — to allow Firestore security rules to scope access cleanly. Hierarchy is reconstructed client-side using parentId and level fields.
Collection: users/{uid}
{
  uid:              string,       // Firebase Auth UID (document ID)
  email:            string,
  displayName:      string,
  photoURL:         string,
  plan:             'free' | 'pro' | 'team',
  stripeCustomerId: string,       // Set on first Stripe interaction
  stripeSubId:      string,       // Active subscription ID
  planExpiresAt:    Timestamp,    // Null for active subscriptions
  createdAt:        Timestamp,
  updatedAt:        Timestamp
}
Collection: projects/{projectId}
{
  projectId:        string,       // Auto-generated document ID
  uid:              string,       // Owner UID (for security rules)
  title:            string,       // AI-generated, user-editable
  description:      string,
  voiceTranscript:  string,       // Raw transcript preserved for reprocessing
  status:           'active' | 'archived',
  taskCount:        number,       // Denormalized for display
  completedCount:   number,
  createdAt:        Timestamp,
  updatedAt:        Timestamp
}
Subcollection: projects/{projectId}/tasks/{taskId}
{
  taskId:              string,    // Auto-generated document ID
  projectId:           string,    // Denormalized for queries
  parentId:            string | null,  // null = Level 1 epic
  level:               1 | 2,     // 1 = epic, 2 = subtask
  title:               string,
  description:         string,
  priority:            'low' | 'medium' | 'high',
  status:              'todo' | 'in_progress' | 'done',
  aiGenerated:         boolean,
  lockedFromRecalib:   boolean,   // User pins task — AI never modifies
  order:               number,    // Display sort order within parent
  createdAt:           Timestamp,
  updatedAt:           Timestamp
}
Subcollection: projects/{projectId}/kb_items/{kbId}
{
  kbId:           string,
  projectId:      string,
  type:           'pdf' | 'url' | 'text',
  label:          string,         // User-facing name
  storageRef:     string | null,  // Firebase Storage path (PDFs only)
  sourceUrl:      string | null,  // For URL type
  extractedText:  string,         // Processed plain text for LLM
  status:         'processing' | 'ready' | 'error',
  processedAt:    Timestamp,
  createdAt:      Timestamp
}
Subcollection: projects/{projectId}/recalib_diffs/{diffId}
{
  diffId:         string,
  projectId:      string,
  kbItemId:       string,         // KB item that triggered this recalibration
  status:         'pending_review' | 'accepted' | 'dismissed',
  changes: [
    {
      taskId:     string,
      field:      'title' | 'description' | 'priority',
      oldValue:   string,
      newValue:   string,
      reason:     string          // LLM-generated explanation
    }
  ],
  newTasks: [TaskModel],          // Net-new tasks suggested by recalibration
  createdAt:      Timestamp
}
4. Firestore Security Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can only read/write their own document
    match /users/{uid} {
      allow read, write: if request.auth.uid == uid;
    }
    // Projects: owner-only access
    match /projects/{projectId} {
      allow read, write: if request.auth.uid == resource.data.uid;
      allow create: if request.auth.uid == request.resource.data.uid;
      // Tasks, KB items, diffs: inherit project ownership
      match /tasks/{taskId} {
        allow read, write: if request.auth.uid ==
          get(/databases/$(database)/documents/projects/$(projectId)).data.uid;
      }
      match /kb_items/{kbId} {
        allow read, write: if request.auth.uid ==
          get(/databases/$(database)/documents/projects/$(projectId)).data.uid;
      }
      match /recalib_diffs/{diffId} {
        allow read, write: if request.auth.uid ==
          get(/databases/$(database)/documents/projects/$(projectId)).data.uid;
      }
    }
  }
}

Task Decomposition Prompt Contract (shared/prompts.ts)
The decomposition prompt must return a strict JSON schema. Cloud Function validates the schema before writing to Firestore.
export const DECOMPOSE_PROMPT = (transcript: string) => `
You are a task decomposition engine for a product manager.
Given a voice transcript of their thoughts, extract a structured project.
Return ONLY valid JSON matching this exact schema:
{
  "projectTitle": string,
  "projectDescription": string,  // 1-2 sentence summary
  "epics": [
    {
      "title": string,
      "description": string,
      "priority": "low" | "medium" | "high",
      "subtasks": [
        {
          "title": string,
          "description": string,
          "priority": "low" | "medium" | "high"
        }
      ]
    }
  ]
}
Rules:
- Generate 2-5 epics maximum
- Generate 2-4 subtasks per epic
- Be specific and actionable — not vague
- Do not invent work not mentioned in the transcript
- If the transcript is unclear, generate fewer, broader tasks
Transcript: ${transcript}
`;
Recalibration Prompt Contract
export const RECALIBRATE_PROMPT = (tasks: Task[], kbText: string) => `
You are a task recalibration engine.
Given a list of incomplete tasks and a new knowledge base document,
suggest improvements ONLY where the document provides relevant new context.
Return ONLY valid JSON:
{
  "changes": [
    {
      "taskId": string,
      "field": "title" | "description" | "priority",
      "newValue": string,
      "reason": string   // 1 sentence explaining why
    }
  ],
  "newTasks": [         // Only if document reveals missing work
    {
      "parentId": string | null,
      "level": 1 | 2,
      "title": string,
      "description": string,
      "priority": "low" | "medium" | "high",
      "reason": string
    }
  ]
}
Rules:
- NEVER modify tasks unless the document provides clear justification
- NEVER suggest changes to completed or locked tasks
- Keep changes minimal and targeted — do not rewrite everything
- reason must reference the document specifically
Incomplete Tasks: ${JSON.stringify(tasks)}
Knowledge Base Document: ${kbText}
`;

taskEditNotifier — Optimistic Update Pattern
// providers/task_edit_notifier.dart
class TaskEditNotifier extends StateNotifier<TaskEditState> {
  final FirestoreService _firestore;
  final String taskId;
  // Step 1: Update local state immediately (no await)
  void updateTitle(String newTitle) {
    state = state.copyWith(title: newTitle, isDirty: true);
  }
  // Step 2: Persist to Firestore — rollback on failure
  Future<void> save() async {
    final previous = state;
    state = state.copyWith(isSaving: true);
    try {
      await _firestore.updateTask(taskId, state.toMap());
      state = state.copyWith(isSaving: false, isDirty: false);
    } catch (e) {
      state = previous.copyWith(isSaving: false, error: e.toString());
    }
  }
}
voiceRecordingProvider — State Machine
enum VoiceStatus { idle, recording, processing, done, error }
class VoiceState {
  final VoiceStatus status;
  final String? transcript;
  final String? newProjectId;
  final String? errorMessage;
}
class VoiceRecordingNotifier extends StateNotifier<VoiceState> {
  Future<void> startRecording() async { ... }   // idle → recording
  Future<void> stopAndProcess() async {         // recording → processing → done
    state = VoiceState(status: VoiceStatus.processing);
    try {
      final result = await _functions.call('processVoice', { audioBase64, mimeType });
      state = VoiceState(status: VoiceStatus.done, newProjectId: result.projectId);
    } catch (e) {
      state = VoiceState(status: VoiceStatus.error, errorMessage: e.toString());
    }
  }
  void reset() => state = VoiceState(status: VoiceStatus.idle);
}

11. Plan Enforcement Logic
Plan limits are enforced exclusively inside Cloud Functions. The Flutter client reads the plan from Firestore for UI gating (hiding buttons, showing upgrade prompts) but never trusts itself to authorise an action.
// shared/planGuard.ts
export async function checkPlanLimit(
  uid: string,
  action: 'voice_input' | 'kb_upload' | 'create_project'
): Promise<void> {
  const userDoc = await db.collection('users').doc(uid).get();
  const { plan, voiceInputsThisMonth } = userDoc.data()!;
  if (action === 'voice_input' && plan === 'free') {
    if (voiceInputsThisMonth >= 5) {
      throw new HttpsError('resource-exhausted',
        'Free plan limit reached. Upgrade to Pro for unlimited voice inputs.');
    }
  }
  if (action === 'kb_upload' && plan === 'free') {
    throw new HttpsError('permission-denied',
      'KB uploads require a Pro plan.');
  }
}
